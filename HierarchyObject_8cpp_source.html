369 double HierarchyObject::getDLambdaUncertainty(int loops) const
370 {
371  if (loops >= 0 && loops <= 2) {
372  return 0.;
373  }
374  if (loops == 3) {
375  return std::abs(dLambdaXtUncertainty) +
376  std::abs(getDLambdaEFT() - getDLambdaH3m());
377  }
378 
379  throw std::runtime_error("Δλ uncertainty " + std::to_string(loops) +
380  " loop(s) is not available.");
381 }

570  * Prints out all information of the HierarchyObject

572 std::ostream& operator<<(std::ostream& ostr, const HierarchyObject& ho)
573 {
574  const int suitableHierarchy = ho.getSuitableHierarchy();
575  const std::string renSchemeString =
576  (toRenSchemes(ho.getRenormalizationScheme()) == mh2_eft::RenSchemes::H3m ||
577  toRenSchemes(ho.getRenormalizationScheme()) == mh2_eft::RenSchemes::H3mMDRBAR)
578  ? "H3m scheme" : "DR'";
579  const std::string massString = ho.getIsAlphab() ? "Msbottom" : "Mstop";
580  const std::string spaces = ho.getIsAlphab() ? " " : " ";
581  ostr << "===================================\n"
582  << "Himalaya HierarchyObject parameters\n"
583  << "===================================\n"
584  << "Ren. scheme = " << renSchemeString << "\n"
585  << "Hierarchy = " << suitableHierarchy << " (" << ho.getH3mHierarchyNotation(suitableHierarchy) << ")\n"
586  << massString << "_1" << spaces << "= " << ho.getMDRMasses()(0) << " GeV (MDR')\n"
587  << massString << "_2" << spaces << "= " << ho.getMDRMasses()(1) << " GeV (MDR')\n"
588  << "Abs. diff 2L = " << ho.getAbsDiff2L() << " GeV\n"
589  << "Rel. diff 2L = " << ho.getRelDiff2L()*100 << " %\n"
590  << "Mh^2_0L = {{" << ho.getDMh(0).row(0)(0) << ", " << ho.getDMh(0).row(0)(1)
591  << "}, {" << ho.getDMh(0).row(1)(0) << ", " << ho.getDMh(0).row(1)(1) << "}} GeV^2\n"
592  << "ΔMh^2_1L = {{" << ho.getDMh(1).row(0)(0) << ", " << ho.getDMh(1).row(0)(1)
593  << "}, {" << ho.getDMh(1).row(1)(0) << ", " << ho.getDMh(1).row(1)(1) << "}} GeV^2\n"
594  << "ΔMh^2_2L = {{" << ho.getDMh(2).row(0)(0) << ", " << ho.getDMh(2).row(0)(1)
595  << "}, {" << ho.getDMh(2).row(1)(0) << ", " << ho.getDMh(2).row(1)(1) << "}} GeV^2\n"
596  << "ΔMh^2_3L = {{" << ho.getDMh(3).row(0)(0) << ", " << ho.getDMh(3).row(0)(1)
597  << "}, {" << ho.getDMh(3).row(1)(0) << ", " << ho.getDMh(3).row(1)(1) << "}} GeV^2\n"
598  << "Exp. uncert. 1L = " << ho.getDMhExpUncertainty(1) << " GeV\n"
599  << "Exp. uncert. 2L = " << ho.getDMhExpUncertainty(2) << " GeV\n"
600  << "Exp. uncert. 3L = " << ho.getDMhExpUncertainty(3) << " GeV\n"
601  << "DR' -> MDR' shift = {{" << ho.getDMhDRbarPrimeToMDRbarPrimeShift().row(0)(0) << ", " << ho.getDMhDRbarPrimeToMDRbarPrimeShift().row(0)(1)
602  << "}, {" << ho.getDMhDRbarPrimeToMDRbarPrimeShift().row(1)(0) << ", " << ho.getDMhDRbarPrimeToMDRbarPrimeShift().row(1)(1) << "}} GeV^2\n"
