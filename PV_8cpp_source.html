37 std::complex<T> fast_log(const std::complex<T>& z) noexcept
38 {
39  const T rz = std::real(z);

46 double fB(const std::complex<double>& x) noexcept
47 {
48  const double re = std::real(x);
49  const double im = std::imag(x);
50 
51  if ((std::abs(re) == 0.0 || std::abs(re) == 1.0) && im == 0.0) {
52  return -1.0;
53  }
54 
55  return std::real(-1.0 + fast_log(1.0 - x) - x*fast_log(1.0 - 1.0/x));
56 }

95 /**
96  * Re(B0(s,x,x,q2)), Eq.(2.4) from [hep-ph/0701051]
97  *
98  * @param p2 squared momentum
99  * @param m2 squared mass
100  * @param q2 squared renormalization scale
101  *
102  * @return Re(B0(s,x,x,q2))
103  */
104 double b0xx(double p2, double m2, double q2) noexcept
105 {
106  if (is_zero(p2, EPSTOL) && is_zero(m2, EPSTOL)) {
107  return 0;
108  }
109 
110  p2 = std::abs(p2);
111  m2 = std::abs(m2);
112  q2 = std::abs(q2);
113 
114  if (p2 < 1e-15) {
115  return -std::log(m2 / q2);
116  }
117 
118  if (m2 < 1e-15*p2) {
119  return 2 - std::log(p2 / q2);
120  }
121 
122  if (m2 < 1e-2*p2) {
123  const double d = m2 / p2;
124  const double logd = std::log(d);
125  return 2 - std::log(p2 / q2)
126  + d * (2 * (1 - logd)
127  + d * (-1 - 2 * logd
128  + d * (-10./3 - 4 * logd
129  + d * (-59./6 - 10 * logd))));
130  }
131 
132  if (is_equal(p2, m2, 1e-12)) {
133  return 0.18620063576578215 - std::log(m2 / q2); // 2 - Pi/Sqrt[3]
134  }
135 
136  if (p2 <= 4 * m2) {
137  return 2 - std::log(m2 / q2) -
138  2 * std::sqrt(4 * m2 / p2 - 1) *
139  std::asin(std::sqrt(p2 / (4 * m2)));
140  }
141 
142  const double sq = std::sqrt(1 - 4 * m2 / p2);
143 
144  // s > 4*m2
145  return 2 - std::log(m2 / q2) +
146  sq * std::log(p2 * (1 - sq) / (2 * m2) - 1);
147 }

150 /**
151  * B0 function with squared arguments, from hep-ph/9606211.
152  * @note returns only the real part of B0
153  */
154 double b0(double p2, double m12, double m22, double q2) noexcept
155 {
156  // protect against infrared divergence
157  if (is_zero(p2, EPSTOL) && is_zero(m12, EPSTOL) && is_zero(m22, EPSTOL)) {
158  return 0;
159  }
160 
161  p2 = std::abs(p2);
162  m12 = std::abs(m12);
163  m22 = std::abs(m22);
164  q2 = std::abs(q2);
165 
166  if (is_close(m12, m22, EPSTOL)) {
167  return b0xx(p2, m12, q2);
168  }
169 
170  if (m12 > m22) {
171  std::swap(m12, m22);
172  }
173 
174  // p2 is no 0
175  if (p2 > 1e-10*m22) {
176  const double s = p2 - m22 + m12;
177  const std::complex<double> imin(m12, -EPSTOL);
178  const std::complex<double> x = std::sqrt(pow2(s) - 4 * p2 * imin);
179  const std::complex<double> xp = (s + sign(s)*x) / (2*p2);
180  const std::complex<double> xm = imin / (xp*p2);
181 
182  return -std::log(p2 / q2) - fB(xp, xm);
183  }
184 
185  if (is_close(m12, m22, EPSTOL)) {
186  return -std::log(m12 / q2);
187  }
188 
189  if (m12 < 1e-30) {
190  return 1 - std::log(m22 / q2);
191  }
192 
193  return 1 - std::log(m22/q2)
194  + m12 * std::log(m22/m12) / (m12 - m22);
195 }

198 /**
199  * Derivative of B0(p^2, m1^2, m2^2, Q^2) w.r.t. p^2, for p^2 = 0.
200  *
201  * @note Implemented only in the p^2 = 0 limit.
202  *
203  * @param m12 squared mass
204  * @param m22 squared mass
205  *
206  * @return derivative of B0 w.r.t. p^2 at p^2 = 0
207  */
208 double d1_b0(double m12, double m22) noexcept
209 {
210  m12 = std::abs(m12);
211  m22 = std::abs(m22);
212 
213  if ((m12 < 0.0001) != (m22 < 0.0001)) {
214  return (m12 - m22) * (m12 + m22) / (2 * pow3(m12 - m22));
215  } else if (m12 < 0.0001 && m22 < 0.0001) {
216  return 0;
217  } else if (std::abs(m22 - m12) < 0.001) {
218  return 1 / (6 * m12) + (m12 - m22) / (12 * pow2(m12));
219  }
220 
221  return ((m12 - m22) * (m12 + m22) + 2 * m12 * m22 * std::log(m22 / m12)) /
222  (2 * pow3(m12 - m22));
223 }
