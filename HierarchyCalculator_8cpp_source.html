166 /**
167  * Calculates the 3-loop mass matrix and other information of the hierarchy selection process.
168  * @param isAlphab a bool which determines if the returned object is proportinal to alpha_b.
169  * @return A HierarchyObject which holds all information of the calculation.
170  */
171 himalaya::HierarchyObject HierarchyCalculator::calculateDMh3L(bool isAlphab)
172 {
173  HierarchyObject ho (isAlphab);
174 
175  if (isAlphab)
176  INFO_MSG("3-loop threshold correction Δλ not available for O(αb*αs^2)!");
177 
178  const int mdrFlag = 0;
179 
180  // set mdrFlag
181  ho.setMDRFlag(mdrFlag);
182 
183  // compare hierarchies and get the best fitting hierarchy
184  compareHierarchies(ho);
185 
186  // calculate the 3-loop Higgs mass matrix for the obtained hierachy in the (M)DRbar' scheme
187  ho.setDMh(3, calculateHierarchy(ho, 0, 0, 1) + shiftH3mToDRbarPrime(ho));
188 
189  // set the alpha_x contributions
190  ho.setDMh(1, getMt41L(ho, mdrFlag, mdrFlag));
191 
192  // set the alpha_x*alpha_s contributions
193  ho.setDMh(2, getMt42L(ho, mdrFlag, mdrFlag));
194 
195  // estimate the uncertainty of the expansion at 3-loop level
196  ho.setDMhExpUncertainty(3, getExpansionUncertainty(ho,
197  ho.getDMh(0) + ho.getDMh(1)
198  + ho.getDMh(2), 0, 0, 1));
199 
200  // set the uncertainty of the expansion at 1-loop level to 0 by default,
201  // if the user needs this value getExpansionUncertainty should be called
202  ho.setDMhExpUncertainty(1, 0.);
203 
204  // calculate shifts needed to convert DR' to other renormalization schemes,
205  // here one needs it with the minus sign to convert DR -> H3m
206  ho.setDMhDRbarPrimeToH3mShift(-shiftH3mToDRbarPrime(ho));
207 
208  // to obtain delta_lambda one has to divide the difference of the two calculations by v^2
209  const double v2 = calcV2();
210  const double gt = sqrt2*p.Mt/std::sqrt(v2);
211 
212  // calculate delta_lambda @ 3-loop level
213  calcDeltaLambda3L(ho, false);
214 
215  // create a modified parameters struct and construct
216  // Mh2EFTCalculator and ThresholdCalculator
217  auto p_mass_ES = p;
218  p_mass_ES.mu2(2,2) = pow2(p.MSt(0));
219  p_mass_ES.mq2(2,2) = pow2(p.MSt(1));
220  himalaya::mh2_eft::Mh2EFTCalculator mh2EFTCalculator(p_mass_ES);
221  himalaya::mh2_eft::ThresholdCalculator tc (p_mass_ES);
222  disable_non_as_terms(mh2EFTCalculator); // omit all but O(at*as^n)
223 
224  // mh_eft^2
225  const double mh2_eft = mh2EFTCalculator.getDeltaMh2EFT0Loop();
226  // 1-Loop prefactor at
227  const double pref_1L = oneLoop * pow2(p.Mt * gt);
228  // 2-Loop prefactor at*as
229  const double pref_2L = twoLoop * pow2(p.Mt * gt * p.g3);
230 
231  ho.setDLambda(0, mh2_eft/v2);
232  ho.setDLambda(1, pref_1L*(tc.getThresholdCorrection(
233  mh2_eft::ThresholdCouplingOrders::LAMBDA_AT, mh2_eft::RenSchemes::DRBARPRIME, 1))/v2);
234  ho.setDLambda(2, pref_2L*(tc.getThresholdCorrection(
235  mh2_eft::ThresholdCouplingOrders::LAMBDA_AT_AS, mh2_eft::RenSchemes::DRBARPRIME, 1))/v2);
236  ho.setDLambda(3, ho.getDLambdaEFT());
237  ho.setDLambdaDRbarPrimeToMSbarShift(0, 0.);
238  ho.setDLambdaDRbarPrimeToMSbarShift(1, 0.);
239  ho.setDLambdaDRbarPrimeToMSbarShift(2, pref_2L*(-4*ho.getDLambda(1)
240  *tc.getThresholdCorrection(mh2_eft::ThresholdCouplingOrders::YT_AS,
241  mh2_eft::RenSchemes::DRBARPRIME, 1))/v2);
242 
243  himalaya::mh2_fo::MSSM_mass_eigenstates mfo(p);
244  const auto dmh_fo = mfo.calculate_Mh2();
245  ho.setDMh2FO(0, std::get<0>(dmh_fo));
246  ho.setDMh2FO(1, std::get<1>(dmh_fo));
247  ho.setDMh2FO(2, std::get<2>(dmh_fo));
248  ho.setDMh2FO(3, 0); // filled later, see below
249 
250  himalaya::mh2_fo::MSSM_mass_eigenstates mfo_atas(p, true);
251  const auto dmh_fo_atas = mfo_atas.calculate_Mh2();
252  ho.setDMh2FOAt(0, 0.);
253  ho.setDMh2FOAt(1, std::get<1>(dmh_fo_atas));
254  ho.setDMh2FOAt(2, std::get<2>(dmh_fo_atas));
255  ho.setDMh2FOAt(3, 0.); // filled later, see below
256 
257  // fill in results of EFT calculation
258  himalaya::mh2_eft::Mh2EFTCalculator mh2EFTCalculator_full(p);
259  ho.setDMh2EFT(0, mh2EFTCalculator_full.getDeltaMh2EFT0Loop());
260  ho.setDMh2EFT(1, mh2EFTCalculator_full.getDeltaMh2EFT1Loop(1, 1));
261  ho.setDMh2EFT(2, mh2EFTCalculator_full.getDeltaMh2EFT2Loop(1, 1));
262  ho.setDMh2EFT(3, mh2EFTCalculator_full.getDeltaMh2EFT3Loop(1, 1, 0)
263  + ho.getDLambdaEFT()*v2);
264 
265  {
266  auto ho_mdr = ho;
267  ho_mdr.setMDRFlag(1);
268  // calculate the DR to MDR shift with the obtained hierarchy
269  ho_mdr.setDMhDRbarPrimeToMDRbarPrimeShift(calcDRbarToMDRbarShift(ho_mdr, true, true));
270  ho_mdr.setDMh(3, calculateHierarchy(ho_mdr, 0, 0, 1) + shiftH3mToDRbarPrime(ho_mdr));
271  Eigen::Vector2d mdrMasses;
272  mdrMasses(0) = ho_mdr.getMDRMasses()(0);
273  mdrMasses(1) = ho_mdr.getMDRMasses()(1);
274  ho.setMDRMasses(mdrMasses);
275  ho.setDMhDRbarPrimeToMDRbarPrimeShift(ho_mdr.getDMhDRbarPrimeToMDRbarPrimeShift()
276  + ho_mdr.getDMh(3) - ho.getDMh(3));
277  }
278 
279  // perturbatively diagonalize
280  {
281  const auto DMh2 = fs_diagonalize_hermitian_perturbatively(
282  ho.getDMh(0), ho.getDMh(1), ho.getDMh(2), ho.getDMh(3));
283  ho.setDMh2(0, std::get<0>(DMh2)(0));
284  ho.setDMh2(1, std::get<1>(DMh2)(0));
285  ho.setDMh2(2, std::get<2>(DMh2)(0));
286  ho.setDMh2(3, std::get<3>(DMh2)(0));
287  // set 3L FO corrections to perturbatively diagonalized O(at*as^n) corrections
288  ho.setDMh2FO(3,std::get<3>(DMh2)(0));
289  ho.setDMh2FOAt(3,std::get<3>(DMh2)(0));
290  }
291 
292  return ho;
293 }

295 /**
296  * Compares deviation of all hierarchies with the exact two-loop result and returns the hierarchy which minimizes the error.
297  * @param ho a HierarchyObject with constant isAlphab.
298  * @return An integer which is identified with the suitable hierarchy.
299  */
300 int HierarchyCalculator::compareHierarchies(himalaya::HierarchyObject& ho)
301 {
302  using namespace himalaya::hierarchies;
303 
304  // temporarily set flags to truncate the expansion
305  const Temporarily_set<int> setThreeLoop(expansionDepth.at(ExpansionDepth::threeLoop), 0);
306  const Temporarily_set<int> setMst(expansionDepth.at(ExpansionDepth::Mst), 0);
307 
308  double error = -1.;
309  int suitableHierarchy = -1;
310 
311  // sine of 2 times beta
312  const double s2b = std::sin(2*calcBeta());
313  const double tb = calcTanBeta();
314 
315  // tree level Higgs mass matrix
316  Eigen::Matrix2d treelvl;
317  treelvl (0,0) = s2b/2.*(pow2(p.MZ) / tb + pow2(p.MA) * tb);
318  treelvl (1,0) = s2b/2.*(-pow2(p.MZ) - pow2(p.MA));
319  treelvl (0,1) = treelvl (1,0);
320  treelvl (1,1) = s2b/2.*(pow2(p.MZ) * tb + pow2(p.MA) / tb);
321 
322  ho.setDMh(0, treelvl);
323 
324  // compare the exact higgs mass at 2-loop level with the expanded expressions to find a suitable hierarchy
325  for (int hierarchy = Hierarchies::FIRST; hierarchy < Hierarchies::NUMBER_OF_HIERARCHIES; hierarchy++) {
326  // first, check if the hierarchy is suitable to the mass spectrum
327  ho.setSuitableHierarchy(hierarchy);
328 
329  if(isHierarchySuitable(ho)){
330  // calculate the exact 1-loop result (only alpha_t/b)
331  const Eigen::Matrix2d Mt41L = getMt41L(ho, ho.getMDRFlag(), 0);
332 
333  // call the routine of Pietro Slavich to get the alpha_s alpha_t/b corrections with the MDRbar masses
334  const Eigen::Matrix2d Mt42L = getMt42L(ho, ho.getMDRFlag(), 0);
335 
336  // calculate the exact Higgs mass at 2-loop (only up to alpha_s alpha_t/b)
337  const double Mh2l = calcSmallestEigenvalue(treelvl + Mt41L + Mt42L);
338 
339  // calculate the expanded 2-loop expression with the specific hierarchy
340  const double Mh2LExpanded = calcSmallestEigenvalue(treelvl + Mt41L + calculateHierarchy(ho, 0, 1, 0));
341 
342  // estimate the error
343  const double twoLoopError = std::abs((Mh2l - Mh2LExpanded));
344 
345  // estimate the uncertainty of the expansion at 2L
346  const double expUncertainty2L = getExpansionUncertainty(ho, treelvl
347  + Mt41L, 0, 1, 0);
348 
349  // estimate the uncertainty of the expansion at 3L
350  const double expUncertainty3L = getExpansionUncertainty(ho, treelvl
351  + Mt41L + Mt42L, 0, 0, 1);
352 
353  // estimate the uncertainty of the difference of delta_lambda @ 3-loop
354  // normalized to the exact logarithms
355  calcDeltaLambda3L(ho, true);
356  /*const double deltaLambdaUncertainty = std::abs((ho.getDLambdaEFT()
357  - ho.getDLambdaH3m())/(ho.getDLambdaEFT() - ho.getDLambdaNonLog()));*/
358 
359  // add these errors to include the error of the expansion in the comparison
360  const double currError = std::sqrt(pow2(twoLoopError/Mh2l)
361  + pow2(expUncertainty2L/Mh2LExpanded));
362 
363  // If the error is negative, it is the first iteration and
364  // there is no hierarchy which fits better. Otherwise,
365  // compare the current error with the last error and choose
366  // the hierarchy which fits best (lowest error).
367  if (error < 0 || currError < error) {
368  error = currError;
369  suitableHierarchy = hierarchy;
370  ho.setAbsDiff2L(twoLoopError);
371  ho.setRelDiff2L(twoLoopError/Mh2l);
372  ho.setDMhExpUncertainty(2, expUncertainty2L);
373  ho.setDMhExpUncertainty(3, expUncertainty3L);
374  }
375  }
376  }
377  ho.setSuitableHierarchy(suitableHierarchy);
378 
379  return suitableHierarchy;
380 }

382 /**
383  * Checks if a hierarchy is suitable to the given mass spectrum.
384  * @param ho a HierarchyObject with constant isAlphab and a hierarchy candidate.
385  * @returns A bool if the hierarchy candidate is suitable to the given mass spectrum.
386  */
387 bool HierarchyCalculator::isHierarchySuitable(const himalaya::HierarchyObject& ho) const
388 {
389  using namespace himalaya::hierarchies;
390 
391  const double Mst1 = ho.getIsAlphab() ? p.MSb(0) : p.MSt(0);
392  const double Mst2 = ho.getIsAlphab() ? p.MSb(1) : p.MSt(1);
393 
394  // check if the squark mass is the heaviest mass
395  const double delta = 1.3; // allow for an offset of 30%
396  const double Mgl = p.MG;
397  const double Msq = calcMeanMsq();
398 
399  if(Mst2 > delta*Msq) return false;
400  if(Mgl > delta*Msq) return false;
401 
402  switch (ho.getSuitableHierarchy()){
403  case Hierarchies::h3:
404  return Mgl > Mst2;
405  case Hierarchies::h32q2g:
406  return (Mst2 >= Msq) && (Mst2 > Mgl);
407  case Hierarchies::h3q22g:
408  return (Msq > Mst2) && (Mst2 > Mgl);
409  case Hierarchies::h4:
410  return (Mst1 < Msq) && (Mst1 >= Mgl);
411  case Hierarchies::h5:
412  return (Mst2 - Mst1 > 0.1*Mst1) && ((Mgl - Mst1) < std::abs(Mgl - Mst2)) && (Mst2 < Msq) && (Mst1 >= Mgl);
413  case Hierarchies::h5g1:
414  return (Mst2 - Mst1 > 0.1*Mst1) && ((Mgl - Mst1) < std::abs(Mgl - Mst2)) && (Mst2 < Msq) && (Mgl > Mst1);
415  case Hierarchies::h6:
416  return (Mst2 - Mst1 > 0.1*Mst1) && ((Mst2 - Mgl) < std::abs(Mgl - Mst1)) && (Mst2*1.3 < Msq) && (Mst2 >= Mgl);
417  case Hierarchies::h6g2:
418  return (Mst2 - Mst1 > 0.1*Mst1) && ((Mst2 - Mgl) < std::abs(Mgl - Mst1)) && (Mst2*1.3 < Msq) && (Mgl > Mst2);
419  case Hierarchies::h6b:
420  return (Mst2 - Mst1 > 0.1*Mst1) && ((Mst2 - Mgl) < std::abs(Mgl - Mst1)) && (Mst2 >= Msq) && (Mst2 >= Mgl);
421  case Hierarchies::h6b2qg2:
422  return (Mst2 - Mst1 > 0.1*Mst1) && ((Mst2 - Mgl) < std::abs(Mgl - Mst1)) && (Mst2 >= Msq) && (Mgl > Mst2);
423  case Hierarchies::h6bq22g:
424  return (Mst2 - Mst1 > 0.1*Mst1) && ((Mst2 - Mgl) < std::abs(Mgl - Mst1)) && (Msq > Mst2) && (Mst2 >= Mgl);
425  case Hierarchies::h6bq2g2:
426  return (Mst2 - Mst1 > 0.1*Mst1) && ((Mst2 - Mgl) < std::abs(Mgl - Mst1)) && (Msq > Mst2) && (Mgl > Mst2);
427  case Hierarchies::h9:
428  return (Mst2 >= Msq) && ((Mst2 - Mst1) < (Mst1 - Mgl));
429  case Hierarchies::h9q2:
430  return (Msq > Mst2) && ((Mst1 - Mst1) < (Mst1 - Mgl));
431  }
432  return false;
433 }

811 /**
812  * Shifts Msx1 according to the hierarchy to the MDR scheme.
813  * @param ho a HierarchyObject with constant isAlphab.
814  * @param oneLoopFlag an integer flag which is 0 or 1 in order to shift the order O(alpha_s).
815  * @param twoLoopFlag an integer flag which is 0 or 1 in order to shift the order O(alpha_s^2).
816  * @return A double which is the MDR sx_1 mass.
817  */
818 double HierarchyCalculator::shiftMst1ToMDR(const himalaya::HierarchyObject& ho,
819  unsigned oneLoopFlag,
820  unsigned twoLoopFlag) const

833  const double lmMsq = std::log(pow2(p.scale / calcMeanMsq()));

836  * (10 * pow2(Msq) * (-1 + 2 * lmMsq + 2 * z2) + pow2(Mst2)
837  * (-1 + 2 * lmMst2 + 2 * z2))) / (3. * pow2(Mst1));

844  break;

847  twoLoopFlag * pow2(Al4p) * Mgl * (-5 * (67 + 84 * lmMgl - 84 * lmMsq) * pow5(Mgl) - 40 * (43 + 30 * lmMgl - 30 * lmMsq) * pow3(Mgl) * pow2(Msq) +

853  8 * twoLoopFlag * pow2(Al4p) * (-135 * pow2(Msq) + 12 * Dmglst2 * Mgl * (1 - 22 * z2) +
854  pow2(Mgl) * (77 + 135 * lmMgl + 72 * pow2(lmMgl) - 75 * lmMsq -
855  90 * lmMgl * lmMsq + 45 * pow2(lmMsq) + 104 * z2))) / (9. * pow2(Mst1));
856  break;
857  case Hierarchies::h9:
858  Mst1mod = (1 + mdr2mst1ka);
859  break;
860  default:
861  break;
862  }
863 
864  return Mst1 * std::sqrt(Mst1mod);
865 }

1329  * Fills in delta_lambda @ 3L to the given HierarchyObject
1330  * @param ho a HierrachyObject
1331  */
1332 void HierarchyCalculator::calcDeltaLambda3L(himalaya::HierarchyObject& ho, bool omitXtOrders) const
1333 {
1334  // set Xt order truncation for EFT contribution to be consistent with H3m
1335  const int suitableHierarchy = ho.getSuitableHierarchy();
1336  const int xtOrder =
1337  (suitableHierarchy == himalaya::hierarchies::Hierarchies::h3
1338  || suitableHierarchy == himalaya::hierarchies::Hierarchies::h32q2g
1339  || suitableHierarchy == himalaya::hierarchies::Hierarchies::h3q22g
1340  || suitableHierarchy == himalaya::hierarchies::Hierarchies::h9
1341  || suitableHierarchy == himalaya::hierarchies::Hierarchies::h9q2) ? 3 : 4;
1342 
1343  // to obtain delta_lambda one has to divide the difference of the two calculations by v^2
1344  const double v2 = calcV2();
1345  const double gt = sqrt2*p.Mt/std::sqrt(v2);
1346  const double pref = threeLoop * pow2(p.Mt * gt * pow2(p.g3));
1347 
1348  // create a modified parameters struct and construct
1349  // Mh2EFTCalculator and ThresholdCalculator
1350  auto p_mass_ES = p;
1351  p_mass_ES.mu2(2,2) = pow2(p.MSt(0));
1352  p_mass_ES.mq2(2,2) = pow2(p.MSt(1));
1353  himalaya::mh2_eft::Mh2EFTCalculator mh2EFTCalculator(p_mass_ES);
1354  himalaya::mh2_eft::ThresholdCalculator tc (p_mass_ES);
1355  disable_non_as_terms(mh2EFTCalculator); // omit all but O(at*as^n)
1356 
1357  // calculate the (non-)logarithmic part of Mh2 without delta_lambda_3L
1358  //
1359  // The first line is equivalent to 64 * dytas - 84 * pow2(dytas) -
1360  // 24 * dytas2 + catas2 including all log(mu^2/Mst1^2) which are
1361  // also included in the H3m result, and the second line omits these
1362  // log(mu^2/Mst1^2) terms since they originate from SM
1363  // contributions. The non-logarithmic part contains only Xt orders
1364  // up to O(Xt^2) which are also included in H3m so no subtraction
1365  // is needed. Checked.
1366  const double subtractionTermH3m = mh2EFTCalculator.getDeltaMh2EFT3Loop(0,1,0);
1367  const double subtractionTermEFT = mh2EFTCalculator.getDeltaMh2EFT3Loop(0,0,0);
1368 
1369  if(omitXtOrders) tc.setXtOrderOfDeltaLambdaAtAs2(xtOrder);
1370 
1371  // calculate the EFT logs. In the first call we calculate the full reconstructed contribution to delta_lambda_3L
1372  // including all logarithmic contributions. In the second line we subtract all non-logarithmic contributions
1373  // to isolate the logarithmic ones. Checked.
1374  const double eftLogs = pref*(
1375  tc.getThresholdCorrection(
1376  mh2_eft::ThresholdCouplingOrders::LAMBDA_AT_AS2, mh2_eft::RenSchemes::DRBARPRIME, 1)
1377  - tc.getThresholdCorrection(
1378  mh2_eft::ThresholdCouplingOrders::LAMBDA_AT_AS2, mh2_eft::RenSchemes::DRBARPRIME, 0));
1379 
1380  // calculate the non-logarithmic part of delta_lambda @ 3L
1381  const double deltaLambda3LNonLog = pref*(ho.getDLambdaNonLog()
1382  + shiftH3mToDRbarPrimeMh2(ho,0)) - subtractionTermEFT;
1383 
1384  // calculate delta_lambda_H3m
1385  ho.setDLambdaH3m((pref*(ho.getDLambdaH3m()
1386  + shiftH3mToDRbarPrimeMh2(ho,1)) - subtractionTermH3m)/v2);
1387 
1388  // caluclate delta_lambda_EFT
1389  ho.setDLambdaEFT((deltaLambda3LNonLog + eftLogs)/v2);
1390 
1391  // save the non-logarithmic part of delta_lambda @ 3L
1392  ho.setDLambdaNonLog(deltaLambda3LNonLog/v2);
1393 
1394  // calculate DR' -> MS shift for delta_lambda 3L
1395  // this shift generates Xt^5*Log(mu) terms for the EFT expression
1396  ho.setDLambdaDRbarPrimeToMSbarShift(3, pref*tc.getDRbarPrimeToMSbarShift(xtOrder,1,0)/v2);
1397  // If orders of Xt are omitted, we subtract them from delta_lambda_EFT to be at the same order
1398  // as delta_lambda_H3m. This ensures that in the hierarchy selection process we don't compare
1399  // wrong orders of Xt.
1400  if(omitXtOrders){
1401  ho.setDLambdaEFT(ho.getDLambdaEFT() + pref*(tc.getDRbarPrimeToMSbarShift(xtOrder,1,0)
1402  - tc.getDRbarPrimeToMSbarShift(xtOrder,1,1))/v2);
1403  }
1404 
1405  // set the uncertainty of delta_lambda due to missing Xt terms
1406  // to summarize: delta_lambda_H3m is always of the x_t order of the suitable
1407  // hierachy (h3, h9 ~ xt^3, h5, h6, h6b ~ xt^4), whereas delta_lambda_EFT
1408  // the non-logarithmic term is of order of the hierarchy but the logarithmic
1409  // part is of order O(xt^4) and using the shift O(xt^5). Note that the shift
1410  // for delta_lambda_H3m is always of order of the hierarchy as well.
1411  // the logarithms are omitted since one would double count the error when
1412  // combining it with delta_exp
1413  const int xt4Flag = xtOrder == 3 ? 1 : 0;
1414  ho.setDLambdaXtUncertainty(
1415  std::abs(xt4Flag*pref/v2*tc.getDRbarPrimeToMSbarXtTerms(tc.getLimit(), 4, 1)));
1416 }
